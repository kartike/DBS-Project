create table inventory(
PartID int primary key,
PartQty int,
PartName varchar(20),
PartOutCost numeric(7,2)
);

  

create table employee(
employeeID int primary key,
name varchar(20),
mobile int,
email varchar(25),
address varchar(25),
taskcount int,
engg int
);


create table work(
jobid int primary key,
productid int,
custname varchar(20),
address varchar(30),
mobile int ,
email varchar(20),
indate date ,
outdate date,
estcost int ,
employeeid int references employee
);


create table work_part(
jobid int primary key references work,
partid references inventory
);
